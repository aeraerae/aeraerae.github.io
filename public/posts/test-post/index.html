fn miller_robin(n: u64, a: u64) -> bool {
    if a % n == 0 { return true; }
    let mut k = n - 1;
    loop {
        let temp = fexp(a as u128, k as u128, n as u128) as u64;
        if temp == n - 1 { return true; }
        if k & 1 == 1 { return temp == 1 || temp == n-1; }
        k >>= 1;
    }
}
